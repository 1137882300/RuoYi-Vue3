
打成镜像文件
docker build -t ruoyi-vue3 ./
启动镜像容器
docker run -d --name ruoyi-vue3 -p 80:80 --network ruoyi-network ruoyi-vue3

----------
改镜像名
docker tag rouyi-java:latest ruoyi-java:latest

启动镜像容器
docker run -d --name ruoyi-java --network=ruoyi-network -p 8088:8080 ruoyi-java:latest

查看网络
docker inspect ruoyi-java --format='{{.NetworkSettings.Networks}}'

查看mysql在容器内的IP
docker inspect -f '{{range .NetworkSettings.Networks}}{{.IPAddress}}{{end}}' mysql-v2

查看容器的状态
docker ps -f name=mysql-v2

创建网络
docker network create ruoyi-network

查看网络
docker network ls


进入redis命令
docker exec -it redis-v2 bash


docker run -d --name redis-v2 --network ruoyi-network redis
docker run -p 6379:6379 --name redis-v2 --network=ruoyi-network -v F:\docker-home\redis-v2\data:/data  -v F:\docker-home\redis-v2\conf\redis.conf:/etc/redis/redis.conf  -d redis redis-server /etc/redis/redis.conf -- appendonly yes
